		}

			factor = 8
			is_enslaved = yes
			can_take_servant_job = no
			OR = {
				has_trait = trait_syncretic_proles
				has_trait = trait_robot_superconductive

			}
		}
		modifier = {
			factor = 2
			OR = {
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
		}
		modifier = {
			factor = 0.25
			can_take_servant_job = yes
		}
		# low income modifier
		modifier = {
			factor = 250
			exists = planet
			planet = {
				has_available_jobs = "elitetechnician"
			}
			exists = owner
			owner = {
				is_ai = yes
				has_monthly_income = {
					resource = energy
					value < 20
				}
			}
		}
	}
