tr_mt_battleship_adopt = {
	modifier = {
		starbase_shipyard_build_speed_mult = 0.75
	}
}

tr_mt_battleship_finish = {
	modifier = {
		ship_hull_mult = 0.05
		ascension_perks_add = 1
	}
}

tr_mt_battleship_1 = {
	on_enabled = {
		give_technology = {
			tech = tech_mt_new_BATTLESHIP1_section_template
			message = no
		}
		if = {
			limit = {
				NOT = { has_technology = tech_battleships }
			}
			add_tech_progress = {
				tech = tech_battleships
				progress = 0.50
			}
		}
	}
	ai_weight = {
		factor = 10000
	}
}
