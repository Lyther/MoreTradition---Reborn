tr_mt_develop_1 = {
	#突破上限
	#旧：领袖等级上限+1
	#新：领袖达到十级时获得强大特质
	# modifier = {
	# 	leader_skill_levels = 1
	# }
	custom_tooltip = tr_mt_develop_1_desc
	on_enabled = {
		hidden_effect = {
			every_owned_leader = {
				if = {
					limit = {
						has_skill = 10
						not = {
							has_trait = leader_trait_mt_develop_scientist
							has_trait = leader_trait_mt_develop_commander
							has_trait = leader_trait_mt_develop_official
						}
					}
					if = {
						limit = {
							leader_class = scientist
						}
						add_trait = leader_trait_mt_develop_scientist
					}
					if = {
						limit = {
							leader_class = commander
						}
						add_trait = leader_trait_mt_develop_commander
					}
					if = {
						limit = {
							leader_class = official
						}
						add_trait = leader_trait_mt_develop_official
					}
				}
			}
			every_pool_leader = {
				if = {
					limit = {
						has_skill = 10
						not = {
							has_trait = leader_trait_mt_develop_scientist
							has_trait = leader_trait_mt_develop_commander
							has_trait = leader_trait_mt_develop_official
						}
					}
					if = {
						limit = {
							leader_class = scientist
						}
						add_trait = leader_trait_mt_develop_scientist
					}
					if = {
						limit = {
							leader_class = commander
						}
						add_trait = leader_trait_mt_develop_commander
					}
					if = {
						limit = {
							leader_class = official
						}
						add_trait = leader_trait_mt_develop_official
					}
				}
			}
		}
	}
	ai_weight = {
		factor = 10000
	}
}
