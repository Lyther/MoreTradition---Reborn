#################
# NEMESIS ASCENSION PERKS
#################

ap_become_the_crisis = {
	potential = {
		has_nemesis = yes
		NOT = {
			has_ascension_perk = ap_become_the_crisis
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			num_ascension_perks > 0
		}
		custom_tooltip = {
			fail_text = "requires_independence"
			is_subject = no
		}
	}

	on_enabled = {
		custom_tooltip = "required_for_mtr_crisis_ascension_perk"
		hidden_effect = {
			set_ai_personality = became_the_crisis
		}
	}

	ai_weight = {
		factor = 7.5
		modifier = {
			factor = 0
			NOR = {
				is_militarist = yes
				is_xenophobe = yes
				is_gestalt = yes
			}
		}
		modifier = {
			factor = 10
			is_homicidal = yes
		}
		modifier = {
			factor = 2
			has_tradition = tr_supremacy_finish
		}
		modifier = {
			factor = 0
			galaxy_percentage < 0.1
		}
		modifier = {
			factor = 2
			galaxy_percentage > 0.2
		}
		modifier = {
			factor = 0
			count_playable_country = {
				count > 1
				limit = {
					has_ascension_perk = ap_become_the_crisis
				}
			}
		}
		modifier = {
			factor = 0
			blocks_disruptive_gameplay = yes
		}
	}
}

ap_cosmogenesis = {
	potential = {
		has_machine_age_dlc = yes
		NOT = {
			has_ascension_perk = ap_cosmogenesis
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_1"
			num_ascension_perks > 0
		}
		custom_tooltip = {
			fail_text = "requires_independence"
			is_subject = no
		}
	}

	on_enabled = {
		custom_tooltip = "required_for_mtr_crisis_ascension_perk"
		hidden_effect = {
			set_variable = {
				which = cosmogenesis_world_advanced_logic_produced
				value = 0
			}
		}
	}

	ai_weight = {
		factor = 7.5
		modifier = {
			factor = 0
			OR = {
				is_egalitarian = yes
				is_xenophile = yes
			}
		}
		modifier = {
			factor = 2
			has_tradition = tr_discovery_finish
		}
		modifier = {
			factor = 2
			has_ascension_perk = ap_technological_ascendancy
		}
		modifier = {
			factor = 2
			has_ascension_perk = ap_enigmatic_engineering
		}
		modifier = {
			factor = 2
			vassals > 0
			any_subject = {
				any_agreement = {
					has_active_specialization = specialist_scholarium
				}
			}
		}
		modifier = {
			factor = 10
			has_country_flag = whisperers_covenant_confirmed
		}
		modifier = {
			factor = 0
			galaxy_percentage < 0.1
		}
		modifier = {
			factor = 2
			galaxy_percentage > 0.2
		}
	}
}
