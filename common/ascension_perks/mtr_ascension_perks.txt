#################
# MT CRISIS ASCENSION PERKS
#################

ap_mtr_crisis = {
	potential = {
		has_nemesis = yes
		has_machine_age_dlc = yes
		NOT = {
			has_ascension_perk = ap_mtr_crisis
		}
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_ascension_perks_3"
			num_ascension_perks > 2
		}
		custom_tooltip = {
			fail_text = "requires_independence"
			is_subject = no
		}
		custom_tooltip = {
			fail_text = "mtr_crisis_ascension_perk_desc"
			has_ascension_perk = ap_become_the_crisis
			has_ascension_perk = ap_cosmogenesis
			is_ai = no
		}
		has_tradition = tr_mtr_crisis_adopt
	}

	on_enabled = {
		custom_tooltip = "mtr_crisis_ascension_perk_desc"
		hidden_effect = {
			set_ai_personality = became_the_crisis
			set_variable = {
				which = cosmogenesis_world_advanced_logic_produced
				value = 0
			}
		}
		activate_crisis_progression = mtr_crisis_path
	}

	ai_weight = {
		factor = 0
	}
}
