		}
	}

#改造传统：血肉苦弱
country_event = {
	id = mt_flag_event.10
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		if = {
			limit = {
				has_machine_age_dlc = yes
				NOT = { has_origin = origin_cybernetic_creed }
			}
			hidden_effect = {
				country_event = { id = cyber.1 }
			}
		}
		else_if = {
			limit = {
				has_machine_age_dlc = yes
				has_origin = origin_cybernetic_creed
			}
			hidden_effect = {
				country_event = { id = machine_age.100 }
			}
		}
		else = {
			hidden_effect = {
				country_event = { id = utopia.2500 }
			}
		}
	}
}

country_event = {
	id = mt_flag_event.11
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		random_situation = {
			limit = {
				is_situation_type = situation_cyberization_hive
			}
			destroy_situation = this
		}
		random_situation = {
			limit = {
				is_situation_type = situation_cyberization
			}
			destroy_situation = this
		}
		random_situation = {
			limit = {
				is_situation_type = situation_cybernetic_creed_cyberization
			}
			destroy_situation = this
		}
		abort_special_project = {
			type = FLESH_IS_WEAK_PROJECT
		}
	}
}
